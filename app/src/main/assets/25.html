• ما هو عدد ســــــــــور القـــــــــــــرءان ؟ 
ــ مائة وأربعة عشر سـورة ( 114 ) .

• ما هو عدد أجــــــــــزاء القـــــــــــرءان ؟
ــ ثلاثــــــــــون جــــــــــــزءاً (30 ) .

• ما هو عدد أحـــــــــزاب القـــــــــــرءان ؟
ــ ستــــــــــــــــون حــــــــزباً (60 ) .

• ما هو عدد أربــــــــــاع القــــــــــــرءان ؟
ــ مئتــــــــان وأربعون ربعاً (240 ) .

• ما هو عدد الأحــــــزاب فـــــي الجــــزء ؟
ــ فــــــــي كل جــــــــزء حـزبان ( 2 ) .

• ما هو عدد الأربـــــاع فـــــي الجـــــــزء ؟
ــ فــــــــي كل جزء ثمانية أرباع ( 8 ) .

• ما هو عدد الأربـــــاع فـــــي الحــــــزب ؟
ــ فــــــــي كل حزب أربعة أرباع ( 4 ) .

• ما هو عدد آيــــــــــات القــــــــــــــرءان ؟
ــ عـــــــدد آياتـــــــه هو ( 6236 ) .

• هل تعلم ما هى أنصاف القرءان ؟
ــ نعم كل شئ له نصفان إلا القرءان فإن له ثمانية أنصاف هى :-
1. نصفان على حروفه وهى تقسم كالآتى :
أ. النصف الأول على الحروف نهايته الفاء فى قوله تعـالى ( وَلْيَتَلَطَّفْ ) من سورة ( الكهف ) آية ( 19 ).
ب. النصف الثانى على الحروف بدايته الواو فى قوله تعالى ( وَلَا يُشْعِرَنَّ ) من سورة ( الكهف ) آية ( 19 ).
2. نصفان على عدد كلماته وعدد كلماته هو ( 77439 ) تقسم كالآتى :-
أ. النصف الأول نهايته كلمـــة ( وَالْجُلُودِ ) فى الآيــــة ( 20 ) من ســـورة ( الحج ) .
ب. النصف الثانى بدايته كلمــة ( وَلَهُـــــم ) فى أول الآية ( 21 ) من سورة ( الحج ) .
3. نصفان على عدد آياته وعدد آياته هو ( 6236 ) آية تقسم كالآتى :-
أ. النصف الأول نهايته على رأس آية ( 45 ) من سورة ( الشعراء ) .
ب. النصف الثانى بدايته على الآيــــــة ( 46 ) من سورة ( الشعراء ) .
4. نصفان على عدد سوره وعدد سوره هو ( 114 ) سورة تقسم كالآتى :-
أ. النصف الأول آخره ســورة ( الحديــد ) .
ب. النصف الثانى أوله سورة ( المجادلة ) .

• ما هى الحالات الممنوعة فى قراءة القرءان ؟
ــ خمس حالات وهى كالآتى :-
1. التطريب : وهـــو الترتيــــــل علــــــى الألحــــــان الموسيقيــــــة .
2. التحزيـن : وهـــو تـــرك القـــارئ عادتـه وطبعـــه ويأتـــى بوجـــه آخـــر فـــى تلاوتـه .
3. الترجيـع : وهـــو الإتيــــــان بأكثـــر مـــــن وجــــــه للقـــراءات فــــى نفــس واحـــــد .
4. الترعيــد : وهو أن يأتــى القارئ بصوت وكأنه يرتعد من شدة ألم أصابه أو برد ألم بـه .
5. التمويـج : وهـــو رفـــع الصـــوت ثـــم خفضـــه وإعــادة الرفـع والخفـض مـرة أخـرى .

• ما هى صفات الحروف مع السكون فى القرءان ؟
ــ للحروف مع السكون أربع صفات هى :-
1. سكــــــون مــــــع القلقلــــــة وحروفـــه ( قطـــــــب جـــــد ) .
2. سكــــــون مــــــع الصفيــــر وحروفـــه ( س ــ ص ــ ز ) .
3. سكـــــــون مـــع الهمـــــس وحروفــــــه ( حثـه شخـص فسكـت ) .
4. سكون مع حبس جريان النفس وحروفـه ( وذرنى أظل مع غض ) .

• ماهى السور التى بدأت بالأحرف المقطعة ؟
ــ هي تسعة وعشرون سورة وهم ( البقرة ــ آل عمران ــ الأعراف ــ يونس ــ هود ــ يوسف ــ الرعد ــ إبراهيم ــ الحجر ــ مريم ــ الشعراء ــ القصص ــ العنكبوت ــ طه ــ الروم ــ لقمان ــ السجدة ــ يس ــ ص ــ غافر ــ فصلت ــ الشورى ــ النمل ــ الزخرف ــ الدخان ــ الجاثية ــ الأحقاف ــ ق ــ القلم ) .

• ما هو عدد الحـــــروف التـــــى بــــدأت بهـــــا الســـــور ؟
ــ أربعــة عشر حــرفاً ( 14 ) .

• ماهي الحروف التــــــــــــي بـــــــدأت بهـــــــا الســــــــور ؟
ــ هى ( صلْهُ سْحَيْراً مَــنْ قَطَعَكَ ).

• ماهي الحروف التـى لاتمـد في الســـــور التـــــى بدأت بأحـــرف ؟
ــ نعم ( حــــــــرف الألـــــــف ) .

• ماهي الحروف التي تمد مداً طبيعياً في السور التي بدأت بأحرف ؟
ــ هي ( حَـــــــــىٌ طَهَـــر ) .

• هل تعلم ماهي الحروف التي تمد مداً لازماً في الســور التي بدأت بأحرف ؟
ــ هي ( سنقـــــص علمك ) . 

• كم غنــــــة فى أحكام القــــرءان الكـــــــريم ؟
ــ ستة غنن في أحكام القرءان هي :- ( إدغام بغنة ــ الإقلاب ــ الإخفاء الشفوي ــ إدغام المثلين الصغير فى حرفى الميم والنون ــ حرفان الغنة الميم والنون المشددتين ــ الإخفاء الحقيقي ) .

• ماهي حواميم القرءان ؟
ــ نعم هم سور ( غافر ــ فصلت ــ الشوري  ــ الزخرف ــ الدخان ــ الجاثية ــ الأحقاف ) .


• ما هو حكم وصل الحروف المقطعة في أول السورة بالآية التي تليها ؟
ــ نصل الحروف المقطعة مع الآية التي تليها كما هي تنطق في حالة الوقف تنطق في حالة الوصل ما عدا في سورة ( آل عمران ) نظراً لالتقاء الساكنين وفيها وجهان المد والقصرمع فتح الميم .

• ما هى سـور الحمــد فـى القـــرءان ؟
ــ هى ( الفاتحة ــ الأنعام ــ الكهف ــ سبإ ــ فاطر ) .

• ماهي أم القـرءان وقلب القرءان وعروس القرءان وزهر القرءان ؟
ــ أم القرءان ( الفاتحة ) ــ قلب القرءان ( يس ) ــ عروس القرءان ( الرحمن ) ــ زهر القرءان هم ( القيامة ــ البلد ــ المطففين ــ الهمزة ) .

• ما هي الســـورة التي إذا قرأتـــها كأنـــك قرأت القـــرءان عشر مـــرات ؟
ــ هي ( سورة يس ) عن أنس رضى الله عنه قال : قال رسول الله صل الله عليه وآله وسلم : ( إن لكل شئ قلباً وقلب القرءان يس ومن قرأ يس كتب الله له بقراءتها قراءة القرءان عشر مرات ) رواه الترمذى .

• مــا هــى الآيــات التــى جمعــت حــروف اللغة العربية ال ( 29 ) ؟
ــ هما آية ( 29 سورة الفتح ) ــ وآية ( 154 سورة آل عمران ) .

• كيف قسم العلماء سور القرءان الكريم من حيث الطول والقصر ؟
ــ قسم علماء السلف السور القرءانية من حيث الطول والقصر إلى ( سور طوال ــ ومإين ــ ومثانى ــ ومفصل ) وهى كالآتى :-
1. الطـــوال : هم سبعة وهي ( البقرة ــ آل عمران ــ النساء ــ المائدة ــ الأنعام ــ الأعراف ــ التوبة ) .
2. المإيــــن : ما زادت عن مائة آية .
3. المثــانى : أقل من المـــائة بقليل .
4. المفصـل : وهو ما تفصل سوره بسملات كثيرة ويبدأ من سورة ( ق ) كما في الكشاف للإمام الزمخشرى رضى الله عنه , وهذا الأخير قسم إلي ( سور طويلة ــ متوسطة ــ قصيرة ) .

• ما عدد السور التى سميت بأسماء الأنبياء ؟
ــ هم ستة سور ( يونس ــ هود ــ يوسف ــ نوح ــ إبراهيم ــ محمد ) .

• ما عدد السور التى بدأت بالتسبيح ؟
ــ هم سبعة سور ( الإسراء ــ الأعلى ــ التغابن ــ الجمعه ــ الصف ــ الحشر ــ الحديد ) .

• تعلم ما هى السور التى تنتهى بأسمائها ؟
ــ هم ثلاث سور ( المسد ــ الناس ــ الماعون ) .

• ما هى السورة التى بدأت وإنتهت بالتسبيح ؟
ــ نعم سورة ( الحشر ) .

• كم مرة ذكر لفظ الجلالة ( الله ) فى القرءان الكريم ؟
ــ ذكر ( 2707 ) مرة .

• ما هى السورة التى ذكر لفظ الجلالة ( الله ) في كل آية ؟
ــ نعم سورة ( المجادلة ) .

• ما هى السور التى لم ترد فيها لفظ الجلالة ( الله ) ؟
ــ نعم في ثلاث سور متتالية هى ( القمر ــ الرحمن ــ الواقعة ) .

• ما هى أطول سورة في القـرءان ؟
ــ هى سورة البقرة وعدد آياتها ( 286 ) آية .

• ما هى أقصر سورة في القرءان ؟
ــ سورة الكوثر وعدد آياتها ( 3 ) آية .

• ما هى أطول آية في القـــــرءان ؟
ــ هي آية ( الدَّيْن ) من سورة البقرة رقم ( 282 ) .

• ما هى أقصر آية في القـــــرءان ؟
ــ هي ( مُدْهَآمَّتَان ) آية ( 64 ) سورة ( الرحمن ) .

• من هوالصحابي الوحيد الذي ذكر اسمه في القرءان ؟
ــ هو ( سيدنا زيد بن حارثة ) في قوله تعالى ( فَلَمَّا قَضَى زَيْدٌ مِّنْهَا وَطَراً زَوَّجْنَاكَهَا لِكَيْ لَا يَكُونَ عَلَى الْمُؤْمِنِينَ حَرَجٌ فِي أَزْوَاجِ أَدْعِيَائِهِمْ إِذَا قَضَوْا مِنْهُنَّ وَطَراً وَكَانَ أَمْرُ اللَّهِ مَفْعُولاً ) سورة الأحزاب : آية ( 37 )
